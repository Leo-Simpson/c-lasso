275 for lam in path:

277 i += 1 

278 teta = (sp_path[i] - lam) / (sp_path[i] - sp_path[i + 1])

284 BETA = np.array([[BETA0[i]] + list(BETA[i]) for i in range(len(BETA0))])

773# Compute the derivatives of the solution Beta and the derivative of lambda*subgradient thanks to the ODE 

774def derivatives(activity, s, Mat, C, Inv, idr, number_act): 

775 """ 

776 Compute the derivatives of the solution Beta and the derivative of lambda*subgradient 

777 thanks to the following equation : 

778 

779 2A[active].T A[active] beta_dot + C[active].T v_dot = s on active variables 

780 C[active] beta_dot = 0 

781 

782 (lambda s)_dot = 2A[inactive].T A[active] beta_dot on inactive variables 

783 

784 Then : Mat = A[inactive].T A[active] and Inv is the inverse of a matrix of the first equaction. 

785 

786 """ 

787 beta_dot = np.zeros(len(activity)) 

788 beta_dot[activity] = -Inv[:number_act, :number_act].dot(s[activity]) 

789 lam_s_dot = -Mat.dot(beta_dot) 

790 

791 if len(C) > 0: 791 ↛ 796line 791 didn't jump to line 796, because the condition on line 791 was never false

792 v_dot = np.zeros(len(C)) 

793 v_dot[idr] = Inv[number_act:, :number_act].dot(s[activity]) 

794 lam_s_dot += C.T.dot(v_dot) 

795 

796 return (beta_dot, lam_s_dot)

840def h_prime(rho, typ): 

841 if rho > 0 and typ == "R2": 

842 # huber regress 

843 # where grad(h)(r) is : 

844 # r if -rho<r<rho ; -rho if r < -rho ; rho if r > rho 

845 return lambda r: np.minimum(r, rho) + np.maximum(r, -rho) - r 

846 

847 elif typ == "C2": 

848 # huber classification 

849 # where grad(h)(r) is : 

850 # (r-1) if 1>r>rho ; (rho-1) if r<rho ; 0 if r >1 

851 # NF*beta0 = (1-rho)*Nminus + np.sum(y[F]) 

852 return lambda r: np.minimum(np.maximum(r, rho) - 1.0, 0.0) 

853 

854 elif typ == "C1": 

855 # classification 

856 # grad(h)(r) is : 

857 # (r-1) if 1>r ; 0 if r >1 

858 return lambda r: np.minimum(r - 1, 0.0) 

859 

860 else: 

861 # R1 or R3 

862 # grad(h)(r) is: 

863 # r 

864 return lambda r: r
